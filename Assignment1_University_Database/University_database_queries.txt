CREATE DATABASE Assignment1;
USE Assignment1;

CREATE TABLE Advisor
( advisor_id INT PRIMARY KEY AUTO_INCREMENT,
  name VARCHAR(20) NOT NULL, email VARCHAR(50)NOT NULL UNIQUE,
  specialization VARCHAR(20) NOT NULL);
  
CREATE TABLE Student
( student_id INT PRIMARY KEY AUTO_INCREMENT,
  first_name VARCHAR(20) NOT NULL, last_name VARCHAR(20) NOT NULL,
  email VARCHAR(50) NOT NULL UNIQUE, dob DATE NOT NULL, advisor_id INT,
  student_status INT NOT NULL,FOREIGN KEY(advisor_id) REFERENCES Advisor(advisor_id)
  ON DELETE SET NULL);
    
CREATE TABLE Course
( course_id VARCHAR(10) PRIMARY KEY, title VARCHAR(20) NOT NULL,
  description VARCHAR(100) NOT NULL, instructor VARCHAR(20) NOT NULL);
      
CREATE TABLE Studentmapcourse
( student_id INT, course_id VARCHAR(10), enrollment_date DATE NOT NULL,
  PRIMARY KEY( student_id, course_id), FOREIGN KEY(student_id) REFERENCES 
  Student(student_id) ON DELETE CASCADE, FOREIGN KEY(course_id) REFERENCES
  Course(course_id) ON DELETE CASCADE);
